023import net.ljcomputing.sr.persistence.DataSourceTable;
024import net.ljcomputing.sr.persistence.EntityPopulator;
025import net.ljcomputing.sr.persistence.impl.ConnectionPool;
026import net.ljcomputing.sr.persistence.impl.EntityPopulatorImpl;
027
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030
031import java.lang.reflect.Constructor;
032import java.lang.reflect.Field;
033import java.sql.Connection;
034import java.sql.PreparedStatement;
035import java.sql.ResultSet;
036import java.sql.SQLException;
037import java.sql.Statement;
038import java.util.ArrayList;
039import java.util.List;

041/**
042 * Abstract repository implementation.
043 *
044 * @author James G. Willmore
045 * @param <T> the Model associated with the repository
046 */
047public abstract class AbstractRepository<T extends Model>
048    implements ModelRepository<T> {
049  
050  /** The Constant LOGGER. */
051  @SuppressWarnings("unused")
052  private static final Logger LOGGER = LoggerFactory
053      .getLogger(AbstractRepository.class);
054
055  /** The connection pool. */
056  protected final ConnectionPool cp;
057
058  /** The entity populator. */
059  protected final EntityPopulator ep;
060
061  /** The connection. */
062  protected Connection conn;
063
064  /** The prepared statement. */
065  protected PreparedStatement ps;
066
067  /** The associated data source table. */
068  protected DataSourceTable table;
069
070  /**
071   * Instantiates a new abstract repository.
072   *
073   * @param table the table
074   * @throws PersistenceException the persistence exception
075   */
076  public AbstractRepository(final DataSourceTable table)
077          throws PersistenceException {
078    this.table = table;
079    this.cp = ConnectionPool.getInstance();
080    this.ep = new EntityPopulatorImpl();
081    obtainConnection();
082  }
083
084  /**
085   * Gets the constructor.
086   *
087   * @return the constructor
088   * @throws Exception the exception
089   */
090  @SuppressWarnings("unchecked")
091  protected Constructor<T> getConstructor() throws Exception {
092    return (Constructor<T>) table.getModel()
093        .getConstructor(new Class[] { EntityPopulator.class, ResultSet.class });
094  }
095
096  /**
097   * Obtain connection.
098   *
099   * @throws PersistenceException the persistence exception
100   */
101  protected void obtainConnection() throws PersistenceException {
102    conn = cp.getConnection();
103  }
104
105  /**
106   * Obtain prepared statement.
107   *
108   * @param sql the sql
109   * @return the prepared statement
110   * @throws PersistenceException the persistence exception
111   */
112  protected PreparedStatement obtainPreparedStatement(final String sql)
113      throws PersistenceException {
114    try {
115      return conn.prepareStatement(sql);
116    } catch (SQLException exception) {
117      throw new PersistenceException(exception);
118    }
119  }
120
121  /**
122   * Close prepared statement.
123   */
124  protected void closePreparedStatement() {
125    if (null != ps) {
126      try {
127        ps.close();
128      } catch (Exception exception) {
129        // do nothing
130      }
131    }
132  }
133
134  /**
135   * Close connection.
136   */
137  protected void closeConnection() {
138    if (null != conn) {
139      try {
140        conn.close();
141      } catch (Exception exception) {
142        // do nothing
143      }
144    }
145  }
146
147  /**
148   * @see net.ljcomputing.sr.repository.impl.ModelRepository#create(net.ljcomputing.sr.model.Model, java.lang.String[])
149   */
150  public void create(final T model, final String... columns)
151      throws PersistenceException {
152    ResultSet rs = null;
153    int key = -1;
154
155    try {
156      ps = conn.prepareStatement(
157          SqlUtils.buildInsertStatement(table.getTableName(), columns),
158          Statement.RETURN_GENERATED_KEYS);
159
160      for (int c = 0; c < columns.length; c++) {
161        try {
162          ps.setObject(c + 1, getModelValue(model, columns[c]));
163        } catch (Exception e) {
164          throw new SQLException(e);
165        }
166      }
167
168      ps.executeUpdate();
169
170      rs = ps.getGeneratedKeys();
171
172      if (rs.next()) {
173        key = rs.getInt(1);
174      }
175
176      model.setId(key);
177      closePreparedStatement();
178      closeConnection();
179    } catch (Exception exception) {
180      throw new PersistenceException(exception);
181    } finally {
182      if (null != rs) {
183        try {
184          rs.close();
185        } catch (SQLException e) {
186          // do nothing
187        }
188      }
189    }
190  }
191
192  /**
193   * @see net.ljcomputing.sr.repository.impl.ModelRepository#update(net.ljcomputing.sr.model.Model, java.lang.String[])
194   */
195  public void update(final T model, final String... columns)
196      throws PersistenceException {
197    final String primaryKey = "id";
198    final String sql = SqlUtils.buildUpdateStatement(table.getTableName(),
199        primaryKey, columns);
200
201    try {
202      ps = obtainPreparedStatement(sql);
203
204      int column = 1;
205      for (Object value : getModelValues(model, columns)) {
206        ps.setObject(column++, value);
207      }
208
209      ps.setObject(column, model.getId());
210
211      ps.executeUpdate();
212      closePreparedStatement();
213      closeConnection();
214    } catch (SQLException exception) {
215      throw new PersistenceException(exception);
216    } catch (Exception exception) {
217      throw new PersistenceException(exception);
218    }
219  }
220
221  /**
222   * @see net.ljcomputing.sr.repository.impl.ModelRepository#delete(net.ljcomputing.sr.model.Model)
223   */
224  public void delete(final T model) throws PersistenceException {
225    delete(model.getId());
226  }
227
228  /**
229   * @see net.ljcomputing.sr.repository.impl.ModelRepository#delete(java.lang.Integer)
230   */
231  public void delete(final Integer id) throws PersistenceException {
232    final String primaryKey = "id";
233    final String sql = SqlUtils.buildDeleteStatement(table.getTableName(),
234        primaryKey);
235
236    try {
237      ps = obtainPreparedStatement(sql);
238      ps.setObject(1, id);
239      ps.executeUpdate();
240      closePreparedStatement();
241      closeConnection();
242    } catch (SQLException exception) {
243      throw new PersistenceException(exception);
244    }
245  }
246
247  /**
248   * @see net.ljcomputing.sr.repository.impl.ModelRepository#readById(java.lang.Integer)
249   */
250  public T readById(final Integer id) throws PersistenceException {
251    final String sql = "select * from " + table.getTableName() + " where id=?";
252    T model = null;
253    ResultSet rs = null;
254
255    try {
256      ps = obtainPreparedStatement(sql);
257
258      ps.setObject(1, id);
259
260      rs = ps.executeQuery();
261
262      while (rs.next()) {
263        model = getConstructor().newInstance(ep, rs);
264      }
265
266      closePreparedStatement();
267      closeConnection();
268    } catch (Exception exception) {
269      throw new PersistenceException(exception);
270    } finally {
271      if (null != rs) {
272        try {
273          rs.close();
274        } catch (SQLException e) {
275          // do nothing
276        }
277      }
278    }
279
280    return model;
281  }
282
283  /**
284   * @see net.ljcomputing.sr.repository.impl.ModelRepository#readAll()
285   */
286  public List<T> readAll() throws PersistenceException {
287    final String sql = "select * from " + table.getTableName();
288    final List<T> list = new ArrayList<T>();
289    ResultSet rs = null;
290
291    try {
292      ps = obtainPreparedStatement(sql);
293      rs = ps.executeQuery();
294
295      while (rs.next()) {
296        list.add(getConstructor().newInstance(ep, rs));
297      }
298
299      closePreparedStatement();
300      closeConnection();
301    } catch (Exception exception) {
302      throw new PersistenceException(exception);
303    } finally {
304      if (null != rs) {
305        try {
306          rs.close();
307        } catch (SQLException e) {
308          // do nothing
309        }
310      }
311    }
312
313    return list;
314  }
315
316  /**
317   * Gets the model values.
318   *
319   * @param model the model
320   * @param fieldNames the field names
321   * @return the model values
322   * @throws Exception the exception
323   */
324  protected Object[] getModelValues(final T model, final String... fieldNames)
325      throws Exception {
326    final Object[] values = new Object[fieldNames.length];
327
328    for (int f = 0; f < fieldNames.length; f++) {
329      values[f] = getModelValue(model, fieldNames[f]);
330    }
331
332    return values;
333  }
334
335  /**
336   * Gets the model value.
337   *
338   * @param model the model
339   * @param fieldName the field name
340   * @return the model value
341   * @throws Exception the exception
342   */
343  protected Object getModelValue(final T model, final String fieldName)
344      throws Exception {
345    final String newFieldName = StringUtils.toMemberCase(fieldName);
346    final Field field = model.getClass().getDeclaredField(newFieldName);
347    field.setAccessible(true);
348    return field.get(model);
349  }
