017package net.ljcomputing.persistence.impl;
018
019import net.ljcomputing.StringUtils;
020import net.ljcomputing.exception.PersistenceException;
021import net.ljcomputing.persistence.EntityPopulator;
022
023import java.lang.reflect.Field;
024import java.sql.ResultSet;
025import java.sql.ResultSetMetaData;
026import java.sql.SQLException;
027
028/**
029 * Implementation of an entity populator, 
030 * which populates entities from a result set.
031 * 
032 * @author James G. Willmore
033 *
034 */
035public class EntityPopulatorImpl implements EntityPopulator {
036
037  /**
038   * @see net.ljcomputing.persistence.EntityPopulator#populate(java.lang.Object, java.sql.ResultSet)
039   */
040  public void populate(final Object entity, final ResultSet resultSet)
041      throws PersistenceException {
042    try {
043      final int columnCount = getResultSetMetaData(resultSet).getColumnCount() + 1;
044
045      for (int column = 1; column < columnCount; column++) {
046        populateMember(entity, resultSet, column);
047      }
048    } catch (SQLException exception) {
049      throw new PersistenceException(exception);
050    }
051  }
052
053  /**
054   * Gets the result set meta data.
055   *
056   * @param resultSet the result set
057   * @return the result set meta data
058   * @throws PersistenceException the persistence exception
059   */
060  private ResultSetMetaData getResultSetMetaData(final ResultSet resultSet)
061      throws PersistenceException {
062    try {
063      return resultSet.getMetaData();
064    } catch (SQLException exception) {
065      throw new PersistenceException(exception);
066    }
067  }
068
069  /**
070   * Populate the entity member.
071   *
072   * @param entity the entity
073   * @param resultSet the result set
074   * @param column the column
075   * @throws PersistenceException the persistence exception
076   */
077  private void populateMember(final Object entity, final ResultSet resultSet,
078      final int column) throws PersistenceException {
079    try {
080      final ResultSetMetaData rsmd = getResultSetMetaData(resultSet); 
081      final String columnName = rsmd.getColumnName(column);
082      final String fieldName = StringUtils.toMemberCase(columnName);
083      final Class<?> entityClass = entity.getClass();
084      final Field member = getEntityMember(entityClass, fieldName);
085      
086      member.setAccessible(true);
087      member.set(entity, resultSet.getObject(column));
088    } catch (SecurityException exception) {
089      throw new PersistenceException(exception);
090    } catch (IllegalArgumentException exception) {
091      throw new PersistenceException(exception);
092    } catch (IllegalAccessException exception) {
093      throw new PersistenceException(exception);
094    } catch (SQLException exception) {
095      throw new PersistenceException(exception);
096    }
097  }
098
099  /**
100   * Gets the entity member.
101   *
102   * @param entityClass the entity class
103   * @param columnMember the column member
104   * @return the entity member
105   * @throws PersistenceException the persistence exception
106   */
107  private Field getEntityMember(final Class<?> entityClass,
108      final String columnMember) throws PersistenceException {
109    Field field = null;
110
111    for (int f = 0; f < entityClass.getSuperclass()
112        .getDeclaredFields().length; f++) {
113      if (null != field) {
114        break;
115      }
116
117      if (columnMember.equals(
118          entityClass.getSuperclass().getDeclaredFields()[f].getName())) {
119        field = entityClass.getSuperclass().getDeclaredFields()[f];
120      }
121    }
122
123    for (int f = 0; f < entityClass.getDeclaredFields().length; f++) {
124      if (null != field) {
125        break;
126      }
127
128      if (columnMember.equals(entityClass.getDeclaredFields()[f].getName())) {
129        field = entityClass.getDeclaredFields()[f];
130      }
131    }
132
133    return field;
134  }
135}
